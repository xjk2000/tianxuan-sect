## 常见技术栈参考

### Vue 3 + Composition API

```vue
<template>
  <div class="user-profile">
    <el-button :loading="loading" @click="handleSubmit">
      提交
    </el-button>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { ElMessage } from 'element-plus'
import { createUser } from '@/api/user'

// Props 定义
const props = defineProps({
  userId: {
    type: String,
    required: true
  },
  mode: {
    type: String,
    default: 'edit'
  }
})

// Events 定义
const emit = defineEmits(['update', 'close'])

// 状态管理
const loading = ref(false)

// 方法
const handleSubmit = async () => {
  try {
    loading.value = true
    const result = await createUser(props.userId)
    ElMessage.success('创建成功')
    emit('update', result)
  } catch (error) {
    console.error('创建用户失败', error)
    ElMessage.error('创建失败，请稍后重试')
  } finally {
    loading.value = false
  }
}
</script>
```

### React + Hooks

```jsx
import React, { useState } from 'react';
import { message } from 'antd';
import { createUser } from '@/api/user';

// Props 类型定义
interface UserProfileProps {
  userId: string;
  mode?: 'edit' | 'view';
  onUpdate?: (data: any) => void;
  onClose?: () => void;
}

const UserProfile: React.FC<UserProfileProps> = ({
  userId,
  mode = 'edit',
  onUpdate,
  onClose
}) => {
  const [loading, setLoading] = useState(false);

  const handleSubmit = async () => {
    try {
      setLoading(true);
      const result = await createUser(userId);
      message.success('创建成功');
      onUpdate?.(result);
    } catch (error) {
      console.error('创建用户失败', error);
      message.error('创建失败，请稍后重试');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="user-profile">
      <button onClick={handleSubmit} disabled={loading}>
        {loading ? '提交中...' : '提交'}
      </button>
    </div>
  );
};

export default UserProfile;
```

### API 调用示例

```javascript
// src/api/user.js
import request from '@/utils/request'

/**
 * 创建用户
 * @param {string} userId - 用户ID
 * @returns {Promise}
 */
export function createUser(userId) {
  return request({
    url: '/api/users',
    method: 'post',
    data: { userId }
  })
}

/**
 * 获取用户列表
 * @param {Object} params - 查询参数
 * @returns {Promise}
 */
export function getUserList(params) {
  return request({
    url: '/api/users',
    method: 'get',
    params
  })
}
```

### 错误处理示例

```javascript
// 统一错误处理
const handleApiError = (error, defaultMessage = '操作失败') => {
  if (error.response) {
    // 服务器返回错误
    const { status, data } = error.response;
    if (status === 400) {
      ElMessage.error(data.message || '请求参数错误');
    } else if (status === 401) {
      ElMessage.error('未登录或登录已过期');
      // 跳转到登录页
    } else if (status === 403) {
      ElMessage.error('无权限访问');
    } else if (status === 500) {
      ElMessage.error('服务器异常，请稍后重试');
    } else {
      ElMessage.error(data.message || defaultMessage);
    }
  } else if (error.request) {
    // 请求发送但未收到响应
    ElMessage.error('网络异常，请检查网络连接');
  } else {
    // 其他错误
    ElMessage.error(defaultMessage);
  }
};
```
